{接近前高缩量小K线选股器}

// 变量定义
N1:=20;  // 前高周期
N2:=5;   // 均量周期

// 前期高点
HH:=HHV(H,N1);

// 接近前高条件(95%以上)
NH:=C>=HH*0.95 AND C<=HH*1.02;

// 成交量萎缩条件
VA:=MA(V,N2);           // 5日均量
VL:=V<VA*0.8;           // 今日量小于均量80%
VL2:=V<REF(V,1)*0.9;    // 今日量小于昨日90%

// 小K线条件
KS:=ABS(C-O)/O<0.03;    // 实体小于3%
KS2:=H-L<C*0.05;        // 振幅小于5%

// 股价位置合理
PR:=C>MA(C,10);         // 股价在10日线上方
PR2:=C<MA(C,60)*1.3;    // 不能偏离60日线太远

// 综合条件
NH AND VL AND VL2 AND KS AND KS2 AND PR AND PR2;
